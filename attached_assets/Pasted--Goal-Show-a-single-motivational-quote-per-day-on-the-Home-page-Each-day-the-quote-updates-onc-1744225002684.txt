import { motivationalQuotes } from "@/lib/motivationalQuotes";

function getDailyQuote() {
  const today = new Date().toDateString(); // e.g. "Wed Apr 10 2025"
  const stored = localStorage.getItem("quoteOfTheDay");

  if (stored) {
    const { date, quote } = JSON.parse(stored);
    if (date === today) return quote;
  }

  // Get quote based on day number to keep it consistent
  const index = new Date().getDate() % motivationalQuotes.length;
  const newQuote = motivationalQuotes[index];

  localStorage.setItem(
    "quoteOfTheDay",
    JSON.stringify({ date: today, quote: newQuote })
  );

  return newQuote;
}
